# [방문 길이](https://school.programmers.co.kr/learn/courses/30/lessons/49994)

### Solution

```cpp
int solution(string dirs) {
    int answer = 0;
    int cx = 5;
    int cy = 5;
    
    //visit 설정 11*11*2 / 0 -> 가로 , 1 -> 세로
    vector<vector<vector<bool>>> visit(11,vector<vector<bool>>(11,vector<bool>(2, false)));
    for(int i =0; i < 11; i++){
        visit[i][10][0] = true;
        visit[10][i][1] = true;
    }
    
    //방향 매핑
    map<char, pair<int,int>> dir{{'U',{-1,0}},{'R',{0,1}},{'D',{1,0}},{'L',{0,-1}}};
    
    //이동
    for(int i = 0 ; i < dirs.length() ; i++)
    {
        int nx = cx + dir[dirs[i]].second;
        int ny = cy + dir[dirs[i]].first;
        
        if(nx <0 || nx>=11 || ny <0 || ny >= 11)
            continue;
        
        if(dirs[i] == 'U' ){
            if(visit[ny][nx][1] == false)
                answer++; 
            visit[ny][nx][1] = true;
        }
        else if(dirs[i] == 'L' ){
            if(visit[ny][nx][0] == false)
                answer++; 
            visit[ny][nx][0] = true;
        }
        else if(dirs[i] == 'D' ){
            if(visit[cy][cx][1] == false)
                answer++; 
            visit[cy][cx][1] = true;
        }
        else if(dirs[i] == 'R' ){
            if(visit[cy][cx][0] == false)
                answer++; 
            visit[cy][cx][0] = true;
        }
        
        cx=nx;
        cy=ny;
    }
    
    return answer;
}
```
